def find_factors(n):
    factors = []
    for i in range(1, n + 1):
        if n % i == 0:
            factors.append(i)
    return factors

if __name__ == "__main__":
    # Example usage
    n = int(input("Enter a number to find its factors: "))
    
    if n < 1:
        print("Please enter a positive integer.")
    else:
        result = find_factors(n)
        print(f"The factors of {n} are: {result}")
