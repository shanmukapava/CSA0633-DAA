def min_max(n):
    if not n:
        return
    min_val=max_val=n[0]
    for num in n[1:]:
        if num<min_val:
            min_val=num
        elif num>max_val:
            max_val=num
        else:
            continue
    return min_val,max_val
    
list=[1, 3, 7, 5, 6, 9, 2, 10, 11]
min_val,max_val=min_max(list)
print("Minimum value:",min_val)
print("Maximum value:",max_val)
